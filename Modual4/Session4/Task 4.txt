import 'package:flutter/material.dart';
import 'team_details_screen.dart';

class TeamGridScreen extends StatelessWidget {
  const TeamGridScreen({super.key});

  @override
  Widget build(BuildContext context) {
    final List<Map<String, String>> teams = [
      {
        "name": "Chennai Super Kings",
        "logo": "https://via.placeholder.com/100"
      },
      {
        "name": "Mumbai Indians",
        "logo": "https://via.placeholder.com/100"
      },
      {
        "name": "Royal Challengers Bengaluru",
        "logo": "https://via.placeholder.com/100"
      },
      {
        "name": "Kolkata Knight Riders",
        "logo": "https://via.placeholder.com/100"
      },
      {
        "name": "Rajasthan Royals",
        "logo": "https://via.placeholder.com/100"
      },
      {
        "name": "Sunrisers Hyderabad",
        "logo": "https://via.placeholder.com/100"
      },
      {
        "name": "Delhi Capitals",
        "logo": "https://via.placeholder.com/100"
      },
      {
        "name": "Gujarat Titans",
        "logo": "https://via.placeholder.com/100"
      },
    ];

    return Scaffold(
      appBar: AppBar(
        title: const Text("IPL Teams"),
        backgroundColor: Colors.blue,
      ),
      body: Padding(
        padding: const EdgeInsets.all(10),
        child: GridView.builder(
          itemCount: teams.length,
          gridDelegate: const SliverGridDelegateWithFixedCrossAxisCount(
            crossAxisCount: 2,
            crossAxisSpacing: 10,
            mainAxisSpacing: 10,
          ),
          itemBuilder: (context, index) {
            return Dismissible(
              key: Key(teams[index]["name"]!),

              // Swipe only from left to right
              direction: DismissDirection.startToEnd,

              // Background while swiping
              background: Container(
                alignment: Alignment.centerLeft,
                padding: const EdgeInsets.only(left: 20),
                color: Colors.green,
                child: const Icon(
                  Icons.bookmark_add,
                  color: Colors.white,
                  size: 30,
                ),
              ),

              // Prevent removing the card
              confirmDismiss: (direction) async {
                ScaffoldMessenger.of(context).showSnackBar(
                  const SnackBar(
                    content: Text("Added to Watchlist"),
                    duration: Duration(seconds: 2),
                  ),
                );
                return false;
              },

              child: InkWell(
                onTap: () {
                  Navigator.push(
                    context,
                    MaterialPageRoute(
                      builder: (_) => TeamDetailsScreen(
                        teamName: teams[index]["name"]!,
                      ),
                    ),
                  );
                },
                child: Card(
                  elevation: 4,
                  child: Column(
                    mainAxisAlignment: MainAxisAlignment.center,
                    children: [
                      Image.network(
                        teams[index]["logo"]!,
                        width: 80,
                        height: 80,
                      ),
                      const SizedBox(height: 10),
                      Text(
                        teams[index]["name"]!,
                        textAlign: TextAlign.center,
                        style: const TextStyle(
                          fontSize: 16,
                          fontWeight: FontWeight.bold,
                        ),
                      ),
                    ],
                  ),
                ),
              ),
            );
          },
        ),
      ),
    );
  }
}
