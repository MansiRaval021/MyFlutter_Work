import 'dart:io';

// Function: Celsius to Fahrenheit
double celsiusToFahrenheit(double celsius) {
  return (celsius * 9 / 5) + 32;
}

// Function: Fahrenheit to Celsius
double fahrenheitToCelsius(double fahrenheit) {
  return (fahrenheit - 32) * 5 / 9;
}

void main() {

  print("2. Fahrenheit to Celsius");

  }
}

import 'dart:io';

// Function for ascending sort
List<int> sortAscending(List<int> list) {
  for (int i = 0; i < list.length; i++) {
    for (int j = 0; j < list.length - i - 1; j++) {
      if (list[j] > list[j + 1]) {
        int temp = list[j];
        list[j] = list[j + 1];
        list[j + 1] = temp;
      }
    }
  }
  return list;
}

// Function for descending sort
List<int> sortDescending(List<int> list) {
  for (int i = 0; i < list.length; i++) {
    for (int j = 0; j < list.length - i - 1; j++) {
      if (list[j] < list[j + 1]) {
        int temp = list[j];
        list[j] = list[j + 1];
        list[j + 1] = temp;
      }
    }
  }
  return list;
}

void main() {
  print("Enter numbers separated by space:");
  String input = stdin.readLineSync()!;

  List<int> numbers = input.split(' ').map((e) => int.parse(e)).toList();

  // Copy list for separate sorting
  List<int> ascList = List.from(numbers);
  List<int> descList = List.from(numbers);

  // Sort lists
  sortAscending(ascList);
  sortDescending(descList);

  print("Ascending Order: $ascList");
  print("Descending Order: $descList");
}

class BankAccount {
  String accountNumber;
  String accountHolder;
  double balance;

  // Constructor
  BankAccount(this.accountNumber, this.accountHolder, this.balance);

  // Deposit method
  void deposit(double amount) {
    balance += amount;
    print("Deposited: $amount");
  }

  // Withdraw method (no overdraft allowed)
  void withdraw(double amount) {
    if (amount <= balance) {
      balance -= amount;
      print("Withdrawn: $amount");
    } else {
      print("Insufficient balance! Withdrawal not allowed.");
    }
  }

  // Check balance method
  void checkBalance() {
    print("Current Balance: $balance");
  }
}

void main() {
  // Create account
  BankAccount acc = BankAccount("123456", "Mansi", 5000);

  acc.checkBalance();

  acc.deposit(2000);
  acc.checkBalance();

  acc.withdraw(3000);
  acc.checkBalance();

  acc.withdraw(5000); // Overdraft attempt
}

    items.add(product);

    for (var item in items) {

  }

  print("Halves: ${processList(numbers, half)}");
}

// Higher-order function
List<double> processList(List<int> list, double Function(int) operation) {

  return list.map((item) => operation(item)).toList();
}

}

// Cube function
double cube(int x) {
  return x * x * x;
}

// Half function
double half(int x) {
  return x / 2;
}
